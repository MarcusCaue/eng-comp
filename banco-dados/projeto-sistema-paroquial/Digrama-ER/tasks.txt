-> Toda Missa deve ter um celebrante principal, seja padre ou bispo

Batisterio {
  - celebrante (RELACIONAMENTO) (X)
  - pessoaBatizada (RELACIONAMENTO) (X)
  - data
  - igreja (RELACIONAMENTO) (X)
  - padrinhos (RELACIONAMENTO) (X)
  - missa (RELACIONAMENTO) (X)
}

Missa {
  - celebrantePrincipal (RELACIONAMENTO) (X)
  - concelebrantes (RELACIONAMENTO) (X)
  - data
  - igreja (RELACIONAMENTO) (X)
  - ofertas
  - categoriaLiturgica
  - nome
  - batisterios (RELACIONAMENTO)
}

Hierarquia Pessoa {
  - id: integer PRIMARY KEY
  - nome: string,
  - idade: integer,
  - batisterio: integer (RELACIONAMENTO)

  Fiel {
    - comunidade: integer (RELACIONAMENTO) (X)
    - pastoral: integer (RELACIONAMENTO) (X)
    - turmaCatequese: integer (RELACIONAMENTO) (X)
  }

  Padre {
    - cargo: string
    - dataOrdenacao: date
    - ordemReligiosa: string
    - bispo: integer (RELACIONAMENTO)
  }

  Bispo {
    - diocese: string
    - dataOrdEpis: date
    - padres: integer (RELACIONAMENTO)
  }
}

Igreja {
  - id: integer
  - nome: string
  - padroeiro: string
  - comunidade: integer (RELACIONAMENTO) (X)
  - endereco: string 
  - dataErecaoCanonica: date
  - ehMatriz: boolean
}

TurmaCatequese {
  - id: integer
  - padroeiro: string
  - comunidade: integer (RELACIONAMENTO) (X)
  - fieis: integer (RELACIONAMENTO) (X)
  - catequista: integer (RELACIONAMENTO) (X)
}

Pastoral {
  - id: integer
  - nome: string
  - coordenador: integer (RELACIONAMENTO) (X)
  - fieis: integer (RELACIONAMENTO) (X)
}

Comunidade {
  - id: integer
  - localizacao: string
  - nome: string
  - padroeiro: string
  - dataFundacao: date
  - fieis: integer (RELACIONAMENTO) (X)
  - igrejas: integer (RELACIONAMENTO) (X)
}

Dizimo {
  - id: integer
  - comunidade: integer (RELACIONAMENTO) (X)
  - data: date
  - valor: real
}
